```csharp
public sealed class EmbeddedBrokerFixture : IAsyncLifetime
{
    public int Port { get; private set; }
    public EmbeddedBrokerService Broker { get; private set; } = default!;
    public async Task InitializeAsync()
    {
        Port = BosPortBul();                       // TcpListener(0) ile boş port
        Broker = new EmbeddedBrokerService(Port, keepAliveSec: 2);  // keep-alive kısa: LWT testi hızlı
        await Broker.StartAsync();
    }
    public async Task DisposeAsync() => await Broker.StopAsync();
    public async Task<IMqttClient> ClientAsync(string id, (string topic,string payload)? lwt = null) { /* MQTTnet client */ }
    static int BosPortBul(){ var l=new TcpListener(IPAddress.Loopback,0); l.Start(); var p=((IPEndPoint)l.LocalEndpoint).Port; l.Stop(); return p; }
}
```
